795int wax_simplifyTypeDescription(const char *in, char *out) {
796 int out_index = 0;
797 int in_index = 0;
798
799 while(in[in_index]) {
800 switch (in[in_index]) {
801 case WAX_TYPE_STRUCT'{':
802 case WAX_TYPE_UNION'(':
803 for (; in[in_index] != '='; in_index++); // Eat the name!
804 in_index++; // Eat the = sign
805 break;
806
807 case WAX_TYPE_ARRAY'[':
808 do {
809 out[out_index++] = in[in_index++];
810 } while(in[in_index] != WAX_TYPE_ARRAY_END']');
811 break;
812
813 case WAX_TYPE_POINTER'^': { //get rid of internal stucture parts
814 out[out_index++] = in[in_index++];
815 for (; in[in_index] == '^'; in_index++); // Eat all the pointers
816
817 switch (in[in_index]) {
818 case WAX_TYPE_UNION'(':
819 case WAX_TYPE_STRUCT'{': {
820 in_index++;
821 int openCurlies = 1;
822
823 for (; openCurlies > 1 || (in[in_index] != WAX_TYPE_UNION_END')' && in[in_index] != WAX_TYPE_STRUCT_END'}'); in_index++) {
824 if (in[in_index] == WAX_TYPE_UNION'(' || in[in_index] == WAX_TYPE_STRUCT'{') openCurlies++;
825 else if (in[in_index] == WAX_TYPE_UNION_END')' || in[in_index] == WAX_TYPE_STRUCT_END'}') openCurlies--;
826 }
827 break;
828 }
829 }
830 }
831
832 case WAX_TYPE_STRUCT_END'}':
833 case WAX_TYPE_UNION_END')':
834 case '0':
835 case '1':
836 case '2':
837 case '3':
838 case '4':
839 case '5':
840 case '6':
841 case '7':
842 case '8':
843 case '9':
844 in_index++;
845 break;
846
847 default:
848 out[out_index++] = in[in_index++];
849 break;
850 }
851 }
852
853 out[out_index] = '\0';
854
855 return out_index;
856}

858int wax_errorFunction(lua_State *L) {
859 lua_getfield(L, LUA_GLOBALSINDEX(-10002), "debug");
860 if (!lua_istable(L, -1)(lua_type(L, (-1)) == 5)) {
861 lua_pop(L, 1)lua_settop(L, -(1)-1);
862 return 1;
863 }
864
865 lua_getfield(L, -1, "traceback");
866 if (!lua_isfunction(L, -1)(lua_type(L, (-1)) == 6)) {
867 lua_pop(L, 2)lua_settop(L, -(2)-1);
868 return 1;
869 }
870 lua_remove(L, -2); // Remove debug
871
872 lua_pushvalue(L, -2); // Grab the error string and place it on the stack
873
874 lua_call(L, 1, 1);
875 lua_remove(L, -2); // Remove original error string
876
877 return 1;
878}

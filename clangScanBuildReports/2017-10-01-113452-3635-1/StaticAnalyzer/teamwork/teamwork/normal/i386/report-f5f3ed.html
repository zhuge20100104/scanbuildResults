413static int methodClosure(lua_State *L) {

415

418
419 // If the only arg is 'self' and there is a selector with no args. USE IT!
420 if (lua_gettop(L) == 1 && lua_isstring(L, lua_upvalueindex(2)((-10002)-(2)))) {
421 selectorName = luaL_checkstring(L, lua_upvalueindex(2))(luaL_checklstring(L, (((-10002)-(2))), ((void*)0)));
422 }
423
424 SEL selector = sel_getUid(selectorName);
425 id instance = instanceUserdata->instance;

428 // If init is called on a class, auto-allocate it.

434 NSMethodSignature *signature = [instance methodSignatureForSelector:selector];

469 }

471

476

478

491 }

496 }
497 }
498 }
499

501 }
502
503 return 1;
504}
